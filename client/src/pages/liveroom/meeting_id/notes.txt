    mp.id AS id,                 -- e.g., 'cam-1', 'cam-host'
    u.full_name AS name,         -- e.g., 'Lebo M.'
    CONCAT(SUBSTRING(u.first_name, 1, 1), SUBSTRING(u.last_name, 1, 1)) AS initials,
    mp.role AS role,             -- 'Presenter', 'Participant'
    mp.status AS status,         -- 'Live', 'Online', 'Muted'
    mp.is_sharing AS sharing,    -- true / false (maps to your new sharing boolean)
    FALSE AS isSpeaking          -- Placeholder for client-side dynamic state
FROM
    meeting_participants mp
INNER JOIN 
    users u ON mp.user_id = u.id
WHERE 
    mp.meeting_id = ?
    AND mp.status != 'offline'   -- Only fetch people actively in the room
ORDER BY 
    -- Optional: Keep the Host/Presenter at the top of the list
    CASE WHEN mp.role = 'Presenter' THEN 1 ELSE 2 END,
    u.full_name ASC;
